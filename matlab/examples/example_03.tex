\section*{Parameteric plot}

\begin{matlab}
   syms t
   xt = exp(abs(t)/10).*sin(5*abs(t));
   yt = exp(abs(t)/10).*cos(5*abs(t));
   zt = t;
   h = fplot3(xt,yt,zt,[-10,10],'r');
   view([-45,20]);
   print(gcf,'example_03_fig3.png','-dpng');
\end{matlab}
